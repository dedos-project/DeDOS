162 /* Don't allow the total size of the HTTP headers (including the status
163  * line) to exceed HTTP_MAX_HEADER_SIZE. This check is here to protect
164  * embedders against denial-of-service attacks where the attacker feeds
165  * us a never-ending header that the embedder keeps buffering.
166  *
167  * This check is arguably the responsibility of embedders but we're doing
168  * it on the embedder's behalf because most won't bother and this way we
169  * make the web a little safer. HTTP_MAX_HEADER_SIZE is still far bigger
170  * than any reasonable request or response so this should never affect
171  * day-to-day operation.
172  */
173 #define COUNT_HEADER_SIZE(V) \
174 do { \
175  parser->nread += (V); \
176  if (UNLIKELY(parser->nread > (HTTP_MAX_HEADER_SIZE))) { \
177  SET_ERRNO(HPE_HEADER_OVERFLOW); \
178  goto error; \

494 /* Our URL parser.
495  *
496  * This is designed to be shared by http_parser_execute() for URL validation,
497  * hence it has a state transition + byte-for-byte interface. In addition, it
498  * is meant to be embedded in http_parser_parse_url(), which does the dirty
499  * work of turning state transitions URL components for its API.
500  *
501  * This function should only be invoked with non-space characters. It is
502  * assumed that the caller cares about (and can detect) the transition between
503  * URL and non-URL states by looking for these.
504  */
505 static enum state
506 parse_url_char(enum state s, const char ch)
507 {
508  if (ch == ' ' || ch == '\r' || ch == '\n') {
509  return s_dead;
510  }
511 
512 #if HTTP_PARSER_STRICT
513  if (ch == '\t' || ch == '\f') {
514  return s_dead;
515  }
516 #endif
517 
518  switch (s) {
519  case s_req_spaces_before_url:
520  /* Proxied requests are followed by scheme of an absolute URI (alpha).
521  * All methods except CONNECT are followed by '/' or '*'.
522  */
523 
524  if (ch == '/' || ch == '*') {
525  return s_req_path;
526  }
527 
528  if (IS_ALPHA(ch)) {
529  return s_req_schema;
530  }
531 
532  break;
533 
534  case s_req_schema:
535  if (IS_ALPHA(ch)) {
536  return s;
537  }
538 
539  if (ch == ':') {
540  return s_req_schema_slash;
541  }
542 
543  break;
544 
545  case s_req_schema_slash:
546  if (ch == '/') {
547  return s_req_schema_slash_slash;
548  }
549 
550  break;
551 
552  case s_req_schema_slash_slash:
553  if (ch == '/') {
554  return s_req_server_start;
555  }
556 
557  break;
558 
559  case s_req_server_with_at:
560  if (ch == '@') {
561  return s_dead;
562  }
563 
564  /* FALLTHROUGH */
565  case s_req_server_start:
566  case s_req_server:
567  if (ch == '/') {
568  return s_req_path;
569  }
570 
571  if (ch == '?') {
572  return s_req_query_string_start;
573  }
574 
575  if (ch == '@') {
576  return s_req_server_with_at;
577  }
578 
579  if (IS_USERINFO_CHAR(ch) || ch == '[' || ch == ']') {
580  return s_req_server;
581  }
582 
583  break;
584 
585  case s_req_path:
586  if (IS_URL_CHAR(ch)) {
587  return s;
588  }
589 
590  switch (ch) {
591  case '?':
592  return s_req_query_string_start;
593 
594  case '#':
595  return s_req_fragment_start;
596  }
597 
598  break;
599 
600  case s_req_query_string_start:
601  case s_req_query_string:
602  if (IS_URL_CHAR(ch)) {
603  return s_req_query_string;
604  }
605 
606  switch (ch) {
607  case '?':
608  /* allow extra '?' in query string */
609  return s_req_query_string;
610 
611  case '#':
612  return s_req_fragment_start;
613  }
614 
615  break;
616 
617  case s_req_fragment_start:
618  if (IS_URL_CHAR(ch)) {
619  return s_req_fragment;
620  }
621 
622  switch (ch) {
623  case '?':
624  return s_req_fragment;
625 
626  case '#':
627  return s;
628  }
629 
630  break;
631 
632  case s_req_fragment:
633  if (IS_URL_CHAR(ch)) {
634  return s;
635  }
636 
637  switch (ch) {
638  case '?':
639  case '#':
640  return s;
641  }
642 
643  break;
644 
645  default:
646  break;
647  }
648 
649  /* We should never fall out of the switch above unless there's an error */
650  return s_dead;
651 }

653 size_t http_parser_execute (http_parser *parser,
654  const http_parser_settings *settings,
655  const char *data,
656  size_t len)
657 {
658  char c, ch;
659  int8_t unhex_val;
660  const char *p = data;
661  const char *header_field_mark = 0;
662  const char *header_value_mark = 0;
663  const char *url_mark = 0;
664  const char *body_mark = 0;
665  const char *status_mark = 0;
666  enum state p_state = (enum state) parser->state;
667  const unsigned int lenient = parser->lenient_http_headers;
668 
669  /* We're in an error state. Don't bother doing anything. */
670  if (HTTP_PARSER_ERRNO(parser) != HPE_OK) {
671  return 0;
672  }
673 
674  if (len == 0) {
675  switch (CURRENT_STATE()) {
676  case s_body_identity_eof:
677  /* Use of CALLBACK_NOTIFY() here would erroneously return 1 byte read if
678  * we got paused.
679  */
680  CALLBACK_NOTIFY_NOADVANCE(message_complete);
681  return 0;
682 
683  case s_dead:
684  case s_start_req_or_res:
685  case s_start_res:
686  case s_start_req:
687  return 0;
688 
689  default:
690  SET_ERRNO(HPE_INVALID_EOF_STATE);
691  return 1;
692  }
693  }
694 
695

697  header_field_mark = data;
698  if (CURRENT_STATE() == s_header_value)
699  header_value_mark = data;
700  switch (CURRENT_STATE()) {
701  case s_req_path:
702  case s_req_schema:
703  case s_req_schema_slash:
704  case s_req_schema_slash_slash:
705  case s_req_server_start:
706  case s_req_server:
707  case s_req_server_with_at:
708  case s_req_query_string_start:
709  case s_req_query_string:
710  case s_req_fragment_start:
711  case s_req_fragment:
712  url_mark = data;
713  break;
714  case s_res_status:
715  status_mark = data;
716  break;
717  default:
718  break;
719  }
720 
721  for (p=data; p != data + len; p++) {
722  ch = *p;
723 
724  if (PARSING_HEADER(CURRENT_STATE()))
725  COUNT_HEADER_SIZE(1);
726 
727 reexecute:
728  switch (CURRENT_STATE()) {
729 
730  case s_dead:
731  /* this state is used after a 'Connection: close' message
732  * the parser will error out if it reads another message
733  */
734  if (LIKELY(ch == CR || ch == LF))
735  break;
736 
737  SET_ERRNO(HPE_CLOSED_CONNECTION);
738  goto error;
739 
740  case s_start_req_or_res:
741  {
742  if (ch == CR || ch == LF)
743  break;
744  parser->flags = 0;
745  parser->content_length = ULLONG_MAX;
746 
747  if (ch == 'H') {
748  UPDATE_STATE(s_res_or_resp_H);
749 
750  CALLBACK_NOTIFY(message_begin);
751  } else {
752  parser->type = HTTP_REQUEST;
753  UPDATE_STATE(s_start_req);
754  REEXECUTE();
755  }
756 
757  break;
758  }
759 
760  case s_res_or_resp_H:
761  if (ch == 'T') {
762  parser->type = HTTP_RESPONSE;
763  UPDATE_STATE(s_res_HT);
764  } else {
765  if (UNLIKELY(ch != 'E')) {
766  SET_ERRNO(HPE_INVALID_CONSTANT);
767  goto error;
768  }
769 
770  parser->type = HTTP_REQUEST;
771  parser->method = HTTP_HEAD;
772  parser->index = 2;
773  UPDATE_STATE(s_req_method);
774  }
775  break;
776 
777  case s_start_res:
778  {
779  parser->flags = 0;
780  parser->content_length = ULLONG_MAX;
781 
782  switch (ch) {
783  case 'H':
784  UPDATE_STATE(s_res_H);
785  break;
786 
787  case CR:
788  case LF:
789  break;
790 
791  default:
792  SET_ERRNO(HPE_INVALID_CONSTANT);
793  goto error;
794  }
795 
796  CALLBACK_NOTIFY(message_begin);
797  break;
798  }
799 
800  case s_res_H:
801  STRICT_CHECK(ch != 'T');
802  UPDATE_STATE(s_res_HT);
803  break;
804 
805  case s_res_HT:
806  STRICT_CHECK(ch != 'T');
807  UPDATE_STATE(s_res_HTT);
808  break;
809 
810  case s_res_HTT:
811  STRICT_CHECK(ch != 'P');
812  UPDATE_STATE(s_res_HTTP);
813  break;
814 
815  case s_res_HTTP:
816  STRICT_CHECK(ch != '/');
817  UPDATE_STATE(s_res_http_major);
818  break;
819 
820  case s_res_http_major:
821  if (UNLIKELY(!IS_NUM(ch))) {
822  SET_ERRNO(HPE_INVALID_VERSION);
823  goto error;
824  }
825 
826  parser->http_major = ch - '0';
827  UPDATE_STATE(s_res_http_dot);
828  break;
829 
830  case s_res_http_dot:
831  {
832  if (UNLIKELY(ch != '.')) {
833  SET_ERRNO(HPE_INVALID_VERSION);
834  goto error;
835  }
836 
837  UPDATE_STATE(s_res_http_minor);
838  break;
839  }
840 
841  case s_res_http_minor:
842  if (UNLIKELY(!IS_NUM(ch))) {
843  SET_ERRNO(HPE_INVALID_VERSION);
844  goto error;
845  }
846 
847  parser->http_minor = ch - '0';
848  UPDATE_STATE(s_res_http_end);
849  break;
850 
851  case s_res_http_end:
852  {
853  if (UNLIKELY(ch != ' ')) {
854  SET_ERRNO(HPE_INVALID_VERSION);
855  goto error;
856  }
857 
858  UPDATE_STATE(s_res_first_status_code);
859  break;
860  }
861 
862  case s_res_first_status_code:
863  {
864  if (!IS_NUM(ch)) {
865  if (ch == ' ') {
866  break;
867  }
868 
869  SET_ERRNO(HPE_INVALID_STATUS);
870  goto error;
871  }
872  parser->status_code = ch - '0';
873  UPDATE_STATE(s_res_status_code);
874  break;
875  }
876 
877  case s_res_status_code:
878  {
879  if (!IS_NUM(ch)) {
880  switch (ch) {
881  case ' ':
882  UPDATE_STATE(s_res_status_start);
883  break;
884  case CR:
885  case LF:
886  UPDATE_STATE(s_res_status_start);
887  REEXECUTE();
888  break;
889  default:
890  SET_ERRNO(HPE_INVALID_STATUS);
891  goto error;
892  }
893  break;
894  }
895 
896  parser->status_code *= 10;
897  parser->status_code += ch - '0';
898 
899  if (UNLIKELY(parser->status_code > 999)) {
900  SET_ERRNO(HPE_INVALID_STATUS);
901  goto error;
902  }
903 
904  break;
905  }
906 
907  case s_res_status_start:
908  {
909  MARK(status);
910  UPDATE_STATE(s_res_status);
911  parser->index = 0;
912 
913  if (ch == CR || ch == LF)
914  REEXECUTE();
915 
916  break;
917  }
918 
919  case s_res_status:
920  if (ch == CR) {
921  UPDATE_STATE(s_res_line_almost_done);
922  CALLBACK_DATA(status);
923  break;
924  }
925 
926  if (ch == LF) {
927  UPDATE_STATE(s_header_field_start);
928  CALLBACK_DATA(status);
929  break;
930  }
931 
932  break;
933 
934  case s_res_line_almost_done:
935  STRICT_CHECK(ch != LF);
936  UPDATE_STATE(s_header_field_start);
937  break;
938 
939  case s_start_req:
940  {
941  if (ch == CR || ch == LF)
942  break;
943  parser->flags = 0;
944  parser->content_length = ULLONG_MAX;
945 
946  if (UNLIKELY(!IS_ALPHA(ch))) {
947  SET_ERRNO(HPE_INVALID_METHOD);
948  goto error;
949  }
950 
951  parser->method = (enum http_method) 0;
952  parser->index = 1;
953  switch (ch) {
954  case 'A': parser->method = HTTP_ACL; break;
955  case 'B': parser->method = HTTP_BIND; break;
956  case 'C': parser->method = HTTP_CONNECT; /* or COPY, CHECKOUT */ break;
957  case 'D': parser->method = HTTP_DELETE; break;
958  case 'G': parser->method = HTTP_GET; break;
959  case 'H': parser->method = HTTP_HEAD; break;
960  case 'L': parser->method = HTTP_LOCK; /* or LINK */ break;
961  case 'M': parser->method = HTTP_MKCOL; /* or MOVE, MKACTIVITY, MERGE, M-SEARCH, MKCALENDAR */ break;
962  case 'N': parser->method = HTTP_NOTIFY; break;
963  case 'O': parser->method = HTTP_OPTIONS; break;
964  case 'P': parser->method = HTTP_POST;
965  /* or PROPFIND|PROPPATCH|PUT|PATCH|PURGE */
966  break;
967  case 'R': parser->method = HTTP_REPORT; /* or REBIND */ break;
968  case 'S': parser->method = HTTP_SUBSCRIBE; /* or SEARCH */ break;
969  case 'T': parser->method = HTTP_TRACE; break;
970  case 'U': parser->method = HTTP_UNLOCK; /* or UNSUBSCRIBE, UNBIND, UNLINK */ break;
971  default:
972  SET_ERRNO(HPE_INVALID_METHOD);
973  goto error;
974  }
975  UPDATE_STATE(s_req_method);
976 
977  CALLBACK_NOTIFY(message_begin);
978 
979  break;
980  }
981 
982  case s_req_method:

986  SET_ERRNO(HPE_INVALID_METHOD);
987  goto error;

989

991  if (ch == ' ' && matcher[parser->index] == '\0') {
992  UPDATE_STATE(s_req_spaces_before_url);
993  } else if (ch == matcher[parser->index]) {
994  ; /* nada */

996 
997  switch (parser->method << 16 | parser->index << 8 | ch) {
998 #define XX(meth, pos, ch, new_meth) \
999  case (HTTP_##meth << 16 | pos << 8 | ch): \
1000  parser->method = HTTP_##new_meth; break;
1001 
1002  XX(POST, 1, 'U', PUT)
1003  XX(POST, 1, 'A', PATCH)
1004  XX(POST, 1, 'R', PROPFIND)
1005  XX(PUT, 2, 'R', PURGE)
1006  XX(CONNECT, 1, 'H', CHECKOUT)
1007  XX(CONNECT, 2, 'P', COPY)
1008  XX(MKCOL, 1, 'O', MOVE)
1009  XX(MKCOL, 1, 'E', MERGE)
1010  XX(MKCOL, 1, '-', MSEARCH)
1011  XX(MKCOL, 2, 'A', MKACTIVITY)
1012  XX(MKCOL, 3, 'A', MKCALENDAR)
1013  XX(SUBSCRIBE, 1, 'E', SEARCH)
1014  XX(REPORT, 2, 'B', REBIND)
1015  XX(PROPFIND, 4, 'P', PROPPATCH)
1016  XX(LOCK, 1, 'I', LINK)
1017  XX(UNLOCK, 2, 'S', UNSUBSCRIBE)
1018  XX(UNLOCK, 2, 'B', UNBIND)
1019  XX(UNLOCK, 3, 'I', UNLINK)
1020 #undef XX
1021  default:
1022  SET_ERRNO(HPE_INVALID_METHOD);
1023  goto error;
1024  }

1026  SET_ERRNO(HPE_INVALID_METHOD);
1027  goto error;
1028  }
1029 
1030  ++parser->index;
1031  break;
1032  }

1045  SET_ERRNO(HPE_INVALID_URL);
1046  goto error;
1047  }
1048 
1049  break;
1050  }

1053  case s_req_schema_slash:
1054  case s_req_schema_slash_slash:
1055  case s_req_server_start:
1056  {
1057  switch (ch) {
1058  /* No whitespace allowed here */
1059  case ' ':
1060  case CR:
1061  case LF:
1062  SET_ERRNO(HPE_INVALID_URL);
1063  goto error;
1064  default:
1065  UPDATE_STATE(parse_url_char(CURRENT_STATE(), ch));
1066  if (UNLIKELY(CURRENT_STATE() == s_dead)) {
1067  SET_ERRNO(HPE_INVALID_URL);
1068  goto error;
1069  }
1070  }
1071 
1072  break;
1073  }

1076  case s_req_server_with_at:
1077  case s_req_path:
1078  case s_req_query_string_start:
1079  case s_req_query_string:
1080  case s_req_fragment_start:
1081  case s_req_fragment:
1082  {
1083  switch (ch) {
1084  case ' ':
1085  UPDATE_STATE(s_req_http_start);
1086  CALLBACK_DATA(url);
1087  break;
1088  case CR:
1089  case LF:
1090  parser->http_major = 0;
1091  parser->http_minor = 9;
1092  UPDATE_STATE((ch == CR) ?
1093  s_req_line_almost_done :
1094  s_header_field_start);
1095  CALLBACK_DATA(url);
1096  break;
1097  default:
1098  UPDATE_STATE(parse_url_char(CURRENT_STATE(), ch));
1099  if (UNLIKELY(CURRENT_STATE() == s_dead)) {
1100  SET_ERRNO(HPE_INVALID_URL);
1101  goto error;
1102  }
1103  }
1104  break;
1105  }
1106

1108  switch (ch) {
1109  case 'H':
1110  UPDATE_STATE(s_req_http_H);
1111  break;
1112  case ' ':
1113  break;
1114  default:
1115  SET_ERRNO(HPE_INVALID_CONSTANT);
1116  goto error;
1117  }
1118  break;
1119

1121  STRICT_CHECK(ch != 'T');
1122  UPDATE_STATE(s_req_http_HT);
1123  break;
1124 
1125  case s_req_http_HT:
1126  STRICT_CHECK(ch != 'T');
1127  UPDATE_STATE(s_req_http_HTT);
1128  break;
1129 
1130  case s_req_http_HTT:
1131  STRICT_CHECK(ch != 'P');
1132  UPDATE_STATE(s_req_http_HTTP);
1133  break;
1134 
1135  case s_req_http_HTTP:
1136  STRICT_CHECK(ch != '/');
1137  UPDATE_STATE(s_req_http_major);
1138  break;
1139 
1140  case s_req_http_major:
1141  if (UNLIKELY(!IS_NUM(ch))) {
1142  SET_ERRNO(HPE_INVALID_VERSION);
1143  goto error;
1144  }
1145 
1146  parser->http_major = ch - '0';
1147  UPDATE_STATE(s_req_http_dot);
1148  break;
1149 
1150  case s_req_http_dot:
1151  {
1152  if (UNLIKELY(ch != '.')) {
1153  SET_ERRNO(HPE_INVALID_VERSION);
1154  goto error;
1155  }
1156 
1157  UPDATE_STATE(s_req_http_minor);
1158  break;
1159  }
1160 
1161  case s_req_http_minor:
1162  if (UNLIKELY(!IS_NUM(ch))) {
1163  SET_ERRNO(HPE_INVALID_VERSION);
1164  goto error;
1165  }
1166 
1167  parser->http_minor = ch - '0';
1168  UPDATE_STATE(s_req_http_end);
1169  break;
1170 
1171  case s_req_http_end:
1172  {
1173  if (ch == CR) {
1174  UPDATE_STATE(s_req_line_almost_done);
1175  break;
1176  }
1177 
1178  if (ch == LF) {
1179  UPDATE_STATE(s_header_field_start);
1180  break;
1181  }
1182 
1183  SET_ERRNO(HPE_INVALID_VERSION);
1184  goto error;
1185  break;
1186  }
1187

1189  case s_req_line_almost_done:
1190  {

1193  goto error;
1194  }
1195 
1196  UPDATE_STATE(s_header_field_start);
1197  break;
1198  }
1199 
1200  case s_header_field_start:

1203  UPDATE_STATE(s_headers_almost_done);
1204  break;
1205  }
1206

1211  REEXECUTE();

1215

1218  goto error;
1219  }

1222

1229  break;
1230 
1231  case 'p':
1232  parser->header_state = h_matching_proxy_connection;
1233  break;
1234 
1235  case 't':
1236  parser->header_state = h_matching_transfer_encoding;
1237  break;
1238 
1239  case 'u':
1240  parser->header_state = h_matching_upgrade;
1241  break;
1242 
1243  default:
1244  parser->header_state = h_general;
1245  break;
1246  }
1247  break;
1248  }
1249

1253  for (; p != data + len; p++) {
1254  ch = *p;

1256

1258  break;

1261  case h_general:
1262  break;
1263

1265  parser->index++;

1267  break;
1268 
1269  case h_CO:
1270  parser->index++;
1271  parser->header_state = (c == 'n' ? h_CON : h_general);
1272  break;
1273 
1274  case h_CON:
1275  parser->index++;
1276  switch (c) {
1277  case 'n':
1278  parser->header_state = h_matching_connection;
1279  break;
1280  case 't':
1281  parser->header_state = h_matching_content_length;
1282  break;
1283  default:
1284  parser->header_state = h_general;
1285  break;
1286  }
1287  break;
1288 
1289  /* connection */
1290 
1291  case h_matching_connection:
1292  parser->index++;
1293  if (parser->index > sizeof(CONNECTION)-1
1294  || c != CONNECTION[parser->index]) {
1295  parser->header_state = h_general;
1296  } else if (parser->index == sizeof(CONNECTION)-2) {
1297  parser->header_state = h_connection;
1298  }
1299  break;
1300 
1301  /* proxy-connection */
1302 
1303  case h_matching_proxy_connection:
1304  parser->index++;

1306  || c != PROXY_CONNECTION[parser->index]) {
1307  parser->header_state = h_general;
1308  } else if (parser->index == sizeof(PROXY_CONNECTION)-2) {
1309  parser->header_state = h_connection;
1310  }
1311  break;
1312 
1313  /* content-length */
1314 
1315  case h_matching_content_length:
1316  parser->index++;
1317  if (parser->index > sizeof(CONTENT_LENGTH)-1
1318  || c != CONTENT_LENGTH[parser->index]) {
1319  parser->header_state = h_general;
1320  } else if (parser->index == sizeof(CONTENT_LENGTH)-2) {
1321  parser->header_state = h_content_length;
1322  }
1323  break;
1324 
1325  /* transfer-encoding */
1326 
1327  case h_matching_transfer_encoding:
1328  parser->index++;
1329  if (parser->index > sizeof(TRANSFER_ENCODING)-1
1330  || c != TRANSFER_ENCODING[parser->index]) {
1331  parser->header_state = h_general;
1332  } else if (parser->index == sizeof(TRANSFER_ENCODING)-2) {
1333  parser->header_state = h_transfer_encoding;
1334  }
1335  break;
1336 
1337  /* upgrade */
1338 
1339  case h_matching_upgrade:
1340  parser->index++;
1341  if (parser->index > sizeof(UPGRADE)-1
1342  || c != UPGRADE[parser->index]) {
1343  parser->header_state = h_general;
1344  } else if (parser->index == sizeof(UPGRADE)-2) {
1345  parser->header_state = h_upgrade;
1346  }
1347  break;
1348 
1349  case h_connection:
1350  case h_content_length:
1351  case h_transfer_encoding:
1352  case h_upgrade:
1353  if (ch != ' ') parser->header_state = h_general;
1354  break;
1355 
1356  default:
1357  assert(0 && "Unknown header_state");
1358  break;
1359  }
1360  }
1361 
1362  COUNT_HEADER_SIZE(p - start);
1363 
1364  if (p == data + len) {
1365  --p;
1366  break;
1367  }
1368 
1369  if (ch == ':') {
1370  UPDATE_STATE(s_header_value_discard_ws);
1371  CALLBACK_DATA(header_field);
1372  break;
1373  }
1374 
1375  SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1376  goto error;
1377  }
1378 
1379  case s_header_value_discard_ws:
1380  if (ch == ' ' || ch == '\t') break;
1381 
1382  if (ch == CR) {
1383  UPDATE_STATE(s_header_value_discard_ws_almost_done);
1384  break;
1385  }
1386 
1387  if (ch == LF) {
1388  UPDATE_STATE(s_header_value_discard_lws);
1389  break;
1390  }
1391 
1392  /* FALLTHROUGH */
1393 
1394  case s_header_value_start:
1395  {
1396  MARK(header_value);
1397 
1398  UPDATE_STATE(s_header_value);
1399  parser->index = 0;

1415  }

1427  }
1428

1431  break;

1435  if (c == 'k') {
1436  parser->header_state = h_matching_connection_keep_alive;
1437  /* looking for 'Connection: close' */
1438  } else if (c == 'c') {

1440  } else if (c == 'u') {
1441  parser->header_state = h_matching_connection_upgrade;
1442  } else {

1444  }

1449  break;

1454  }

1462  for (; p != data + len; p++) {

1464  if (ch == CR) {
1465  UPDATE_STATE(s_header_almost_done);
1466  parser->header_state = h_state;
1467  CALLBACK_DATA(header_value);
1468  break;
1469  }
1470 
1471  if (ch == LF) {
1472  UPDATE_STATE(s_header_almost_done);
1473  COUNT_HEADER_SIZE(p - start);
1474  parser->header_state = h_state;
1475  CALLBACK_DATA_NOADVANCE(header_value);
1476  REEXECUTE();
1477  }
1478

1480  SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1481  goto error;
1482  }
1483

1485

1487  case h_general:
1488  {
1489  const char* p_cr;
1490  const char* p_lf;
1491  size_t limit = data + len - p;
1492 
1493  limit = MIN(limit, HTTP_MAX_HEADER_SIZE);
1494

1496  p_lf = (const char*) memchr(p, LF, limit);
1497  if (p_cr != NULL) {
1498  if (p_lf != NULL && p_cr >= p_lf)
1499  p = p_lf;
1500  else
1501  p = p_cr;
1502  } else if (UNLIKELY(p_lf != NULL)) {
1503  p = p_lf;
1504  } else {
1505  p = data + len;
1506  }
1507  --p;
1508

1510  }

1512  case h_connection:
1513  case h_transfer_encoding:
1514  assert(0 && "Shouldn't get here.");
1515  break;
1516

1518  {
1519  uint64_t t;
1520 
1521  if (ch == ' ') break;
1522 
1523  if (UNLIKELY(!IS_NUM(ch))) {
1524  SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1525  parser->header_state = h_state;
1526  goto error;
1527  }
1528 
1529  t = parser->content_length;
1530  t *= 10;
1531  t += ch - '0';
1532 
1533  /* Overflow? Test against a conservative limit for simplicity. */
1534  if (UNLIKELY((ULLONG_MAX - 10) / 10 < parser->content_length)) {
1535  SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1536  parser->header_state = h_state;
1537  goto error;
1538  }
1539 
1540  parser->content_length = t;
1541  break;
1542  }

1545  case h_matching_transfer_encoding_chunked:
1546  parser->index++;
1547  if (parser->index > sizeof(CHUNKED)-1
1548  || c != CHUNKED[parser->index]) {
1549  h_state = h_general;
1550  } else if (parser->index == sizeof(CHUNKED)-2) {
1551  h_state = h_transfer_encoding_chunked;
1552  }
1553  break;
1554

1556  /* looking for 'Connection: keep-alive' */
1557  if (c == 'k') {
1558  h_state = h_matching_connection_keep_alive;
1559  /* looking for 'Connection: close' */
1560  } else if (c == 'c') {
1561  h_state = h_matching_connection_close;
1562  } else if (c == 'u') {
1563  h_state = h_matching_connection_upgrade;
1564  } else if (STRICT_TOKEN(c)) {
1565  h_state = h_matching_connection_token;
1566  } else if (c == ' ' || c == '\t') {
1567  /* Skip lws */
1568  } else {
1569  h_state = h_general;
1570  }
1571  break;
1572 
1573  /* looking for 'Connection: keep-alive' */
1574  case h_matching_connection_keep_alive:
1575  parser->index++;
1576  if (parser->index > sizeof(KEEP_ALIVE)-1
1577  || c != KEEP_ALIVE[parser->index]) {
1578  h_state = h_matching_connection_token;
1579  } else if (parser->index == sizeof(KEEP_ALIVE)-2) {
1580  h_state = h_connection_keep_alive;
1581  }
1582  break;
1583 
1584  /* looking for 'Connection: close' */
1585  case h_matching_connection_close:
1586  parser->index++;
1587  if (parser->index > sizeof(CLOSE)-1 || c != CLOSE[parser->index]) {
1588  h_state = h_matching_connection_token;
1589  } else if (parser->index == sizeof(CLOSE)-2) {
1590  h_state = h_connection_close;
1591  }
1592  break;
1593 
1594  /* looking for 'Connection: upgrade' */
1595  case h_matching_connection_upgrade:
1596  parser->index++;
1597  if (parser->index > sizeof(UPGRADE) - 1 ||
1598  c != UPGRADE[parser->index]) {
1599  h_state = h_matching_connection_token;
1600  } else if (parser->index == sizeof(UPGRADE)-2) {
1601  h_state = h_connection_upgrade;
1602  }
1603  break;
1604 
1605  case h_matching_connection_token:
1606  if (ch == ',') {
1607  h_state = h_matching_connection_token_start;
1608  parser->index = 0;
1609  }
1610  break;
1611 
1612  case h_transfer_encoding_chunked:
1613  if (ch != ' ') h_state = h_general;
1614  break;
1615 
1616  case h_connection_keep_alive:
1617  case h_connection_close:
1618  case h_connection_upgrade:
1619  if (ch == ',') {
1620  if (h_state == h_connection_keep_alive) {
1621  parser->flags |= F_CONNECTION_KEEP_ALIVE;
1622  } else if (h_state == h_connection_close) {
1623  parser->flags |= F_CONNECTION_CLOSE;
1624  } else if (h_state == h_connection_upgrade) {
1625  parser->flags |= F_CONNECTION_UPGRADE;
1626  }
1627  h_state = h_matching_connection_token_start;
1628  parser->index = 0;
1629  } else if (ch != ' ') {
1630  h_state = h_matching_connection_token;
1631  }
1632  break;
1633 
1634  default:
1635  UPDATE_STATE(s_header_value);
1636  h_state = h_general;
1637  break;
1638  }
1639  }
1640  parser->header_state = h_state;
1641 
1642  COUNT_HEADER_SIZE(p - start);
1643 
1644  if (p == data + len)
1645  --p;
1646  break;
1647  }
1648 
1649  case s_header_almost_done:
1650  {
1651  if (UNLIKELY(ch != LF)) {
1652  SET_ERRNO(HPE_LF_EXPECTED);
1653  goto error;
1654  }
1655 
1656  UPDATE_STATE(s_header_value_lws);
1657  break;
1658  }
1659 
1660  case s_header_value_lws:
1661  {
1662  if (ch == ' ' || ch == '\t') {
1663  UPDATE_STATE(s_header_value_start);
1664  REEXECUTE();
1665  }
1666 
1667  /* finished the header */
1668  switch (parser->header_state) {
1669  case h_connection_keep_alive:
1670  parser->flags |= F_CONNECTION_KEEP_ALIVE;
1671  break;
1672  case h_connection_close:
1673  parser->flags |= F_CONNECTION_CLOSE;
1674  break;
1675  case h_transfer_encoding_chunked:
1676  parser->flags |= F_CHUNKED;
1677  break;
1678  case h_connection_upgrade:
1679  parser->flags |= F_CONNECTION_UPGRADE;
1680  break;
1681  default:
1682  break;
1683  }
1684 
1685  UPDATE_STATE(s_header_field_start);
1686  REEXECUTE();
1687  }
1688 
1689  case s_header_value_discard_ws_almost_done:
1690  {
1691  STRICT_CHECK(ch != LF);
1692  UPDATE_STATE(s_header_value_discard_lws);
1693  break;
1694  }
1695 
1696  case s_header_value_discard_lws:
1697  {
1698  if (ch == ' ' || ch == '\t') {
1699  UPDATE_STATE(s_header_value_discard_ws);
1700  break;
1701  } else {
1702  switch (parser->header_state) {
1703  case h_connection_keep_alive:
1704  parser->flags |= F_CONNECTION_KEEP_ALIVE;
1705  break;
1706  case h_connection_close:
1707  parser->flags |= F_CONNECTION_CLOSE;
1708  break;
1709  case h_connection_upgrade:
1710  parser->flags |= F_CONNECTION_UPGRADE;
1711  break;
1712  case h_transfer_encoding_chunked:
1713  parser->flags |= F_CHUNKED;
1714  break;
1715  default:
1716  break;
1717  }
1718 
1719  /* header value was empty */
1720  MARK(header_value);
1721  UPDATE_STATE(s_header_field_start);
1722  CALLBACK_DATA_NOADVANCE(header_value);
1723  REEXECUTE();
1724  }
1725  }
1726 
1727  case s_headers_almost_done:
1728  {
1729  STRICT_CHECK(ch != LF);
1730 
1731  if (parser->flags & F_TRAILING) {
1732  /* End of a chunked request */
1733  UPDATE_STATE(s_message_done);
1734  CALLBACK_NOTIFY_NOADVANCE(chunk_complete);
1735  REEXECUTE();
1736  }
1737 
1738  /* Cannot use chunked encoding and a content-length header together
1739  per the HTTP specification. */
1740  if ((parser->flags & F_CHUNKED) &&
1741  (parser->flags & F_CONTENTLENGTH)) {
1742  SET_ERRNO(HPE_UNEXPECTED_CONTENT_LENGTH);
1743  goto error;
1744  }
1745 
1746  UPDATE_STATE(s_headers_done);
1747 
1748  /* Set this here so that on_headers_complete() callbacks can see it */
1749  if ((parser->flags & F_UPGRADE) &&
1750  (parser->flags & F_CONNECTION_UPGRADE)) {
1751  /* For responses, "Upgrade: foo" and "Connection: upgrade" are
1752  * mandatory only when it is a 101 Switching Protocols response,
1753  * otherwise it is purely informational, to announce support.
1754  */
1755  parser->upgrade =
1756  (parser->type == HTTP_REQUEST || parser->status_code == 101);
1757  } else {
1758  parser->upgrade = (parser->method == HTTP_CONNECT);
1759  }
1760 
1761  /* Here we call the headers_complete callback. This is somewhat
1762  * different than other callbacks because if the user returns 1, we
1763  * will interpret that as saying that this message has no body. This
1764  * is needed for the annoying case of recieving a response to a HEAD
1765  * request.
1766  *
1767  * We'd like to use CALLBACK_NOTIFY_NOADVANCE() here but we cannot, so
1768  * we have to simulate it by handling a change in errno below.
1769  */
1770  if (settings->on_headers_complete) {
1771  switch (settings->on_headers_complete(parser)) {
1772  case 0:
1773  break;
1774 
1775  case 2:
1776  parser->upgrade = 1;
1777 
1778  /* FALLTHROUGH */
1779  case 1:
1780  parser->flags |= F_SKIPBODY;
1781  break;
1782 
1783  default:
1784  SET_ERRNO(HPE_CB_headers_complete);
1785  RETURN(p - data); /* Error */
1786  }
1787  }
1788 
1789  if (HTTP_PARSER_ERRNO(parser) != HPE_OK) {
1790  RETURN(p - data);
1791  }
1792 
1793  REEXECUTE();
1794  }
1795 
1796  case s_headers_done:
1797  {
1798  int hasBody;
1799  STRICT_CHECK(ch != LF);
1800 
1801  parser->nread = 0;
1802 
1803  hasBody = parser->flags & F_CHUNKED ||
1804  (parser->content_length > 0 && parser->content_length != ULLONG_MAX);
1805  if (parser->upgrade && (parser->method == HTTP_CONNECT ||
1806  (parser->flags & F_SKIPBODY) || !hasBody)) {
1807  /* Exit, the rest of the message is in a different protocol. */
1808  UPDATE_STATE(NEW_MESSAGE());
1809  CALLBACK_NOTIFY(message_complete);
1810  RETURN((p - data) + 1);
1811  }
1812 
1813  if (parser->flags & F_SKIPBODY) {
1814  UPDATE_STATE(NEW_MESSAGE());
1815  CALLBACK_NOTIFY(message_complete);
1816  } else if (parser->flags & F_CHUNKED) {
1817  /* chunked encoding - ignore Content-Length header */
1818  UPDATE_STATE(s_chunk_size_start);
1819  } else {
1820  if (parser->content_length == 0) {
1821  /* Content-Length header given but zero: Content-Length: 0\r\n */
1822  UPDATE_STATE(NEW_MESSAGE());
1823  CALLBACK_NOTIFY(message_complete);
1824  } else if (parser->content_length != ULLONG_MAX) {
1825  /* Content-Length header given and non-zero */
1826  UPDATE_STATE(s_body_identity);
1827  } else {
1828  if (!http_message_needs_eof(parser)) {
1829  /* Assume content-length 0 - read the next */
1830  UPDATE_STATE(NEW_MESSAGE());
1831  CALLBACK_NOTIFY(message_complete);
1832  } else {
1833  /* Read body until EOF */
1834  UPDATE_STATE(s_body_identity_eof);
1835  }
1836  }
1837  }
1838 
1839  break;
1840  }
1841 
1842  case s_body_identity:
1843  {
1844  uint64_t to_read = MIN(parser->content_length,
1845  (uint64_t) ((data + len) - p));
1846 
1847  assert(parser->content_length != 0
1848  && parser->content_length != ULLONG_MAX);
1849 
1850  /* The difference between advancing content_length and p is because
1851  * the latter will automaticaly advance on the next loop iteration.
1852  * Further, if content_length ends up at 0, we want to see the last
1853  * byte again for our message complete callback.
1854  */
1855  MARK(body);
1856  parser->content_length -= to_read;
1857  p += to_read - 1;
1858 
1859  if (parser->content_length == 0) {
1860  UPDATE_STATE(s_message_done);
1861 
1862  /* Mimic CALLBACK_DATA_NOADVANCE() but with one extra byte.
1863  *
1864  * The alternative to doing this is to wait for the next byte to
1865  * trigger the data callback, just as in every other case. The
1866  * problem with this is that this makes it difficult for the test
1867  * harness to distinguish between complete-on-EOF and
1868  * complete-on-length. It's not clear that this distinction is
1869  * important for applications, but let's keep it for now.
1870  */
1871  CALLBACK_DATA_(body, p - body_mark + 1, p - data);
1872  REEXECUTE();
1873  }
1874 
1875  break;
1876  }
1877 
1878  /* read until EOF */
1879  case s_body_identity_eof:
1880  MARK(body);
1881  p = data + len - 1;
1882 
1883  break;
1884 
1885  case s_message_done:
1886  UPDATE_STATE(NEW_MESSAGE());
1887  CALLBACK_NOTIFY(message_complete);
1888  if (parser->upgrade) {
1889  /* Exit, the rest of the message is in a different protocol. */
1890  RETURN((p - data) + 1);
1891  }
1892  break;
1893 
1894  case s_chunk_size_start:
1895  {
1896  assert(parser->nread == 1);
1897  assert(parser->flags & F_CHUNKED);
1898 
1899  unhex_val = unhex[(unsigned char)ch];
1900  if (UNLIKELY(unhex_val == -1)) {
1901  SET_ERRNO(HPE_INVALID_CHUNK_SIZE);
1902  goto error;
1903  }
1904 
1905  parser->content_length = unhex_val;
1906  UPDATE_STATE(s_chunk_size);
1907  break;
1908  }
1909 
1910  case s_chunk_size:
1911  {
1912  uint64_t t;
1913 
1914  assert(parser->flags & F_CHUNKED);
1915 
1916  if (ch == CR) {
1917  UPDATE_STATE(s_chunk_size_almost_done);
1918  break;
1919  }
1920 
1921  unhex_val = unhex[(unsigned char)ch];
1922 
1923  if (unhex_val == -1) {
1924  if (ch == ';' || ch == ' ') {
1925  UPDATE_STATE(s_chunk_parameters);
1926  break;
1927  }
1928 
1929  SET_ERRNO(HPE_INVALID_CHUNK_SIZE);
1930  goto error;
1931  }
1932 
1933  t = parser->content_length;
1934  t *= 16;
1935  t += unhex_val;
1936 
1937  /* Overflow? Test against a conservative limit for simplicity. */
1938  if (UNLIKELY((ULLONG_MAX - 16) / 16 < parser->content_length)) {
1939  SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1940  goto error;
1941  }
1942 
1943  parser->content_length = t;
1944  break;
1945  }
1946 
1947  case s_chunk_parameters:
1948  {
1949  assert(parser->flags & F_CHUNKED);
1950  /* just ignore this shit. TODO check for overflow */
1951  if (ch == CR) {
1952  UPDATE_STATE(s_chunk_size_almost_done);
1953  break;
1954  }
1955  break;
1956  }
1957 
1958  case s_chunk_size_almost_done:
1959  {
1960  assert(parser->flags & F_CHUNKED);
1961  STRICT_CHECK(ch != LF);
1962 
1963  parser->nread = 0;
1964 
1965  if (parser->content_length == 0) {
1966  parser->flags |= F_TRAILING;
1967  UPDATE_STATE(s_header_field_start);
1968  } else {
1969  UPDATE_STATE(s_chunk_data);
1970  }
1971  CALLBACK_NOTIFY(chunk_header);
1972  break;
1973  }
1974 
1975  case s_chunk_data:
1976  {
1977  uint64_t to_read = MIN(parser->content_length,
1978  (uint64_t) ((data + len) - p));
1979 
1980  assert(parser->flags & F_CHUNKED);
1981  assert(parser->content_length != 0
1982  && parser->content_length != ULLONG_MAX);
1983 
1984  /* See the explanation in s_body_identity for why the content
1985  * length and data pointers are managed this way.
1986  */
1987  MARK(body);
1988  parser->content_length -= to_read;
1989  p += to_read - 1;
1990 
1991  if (parser->content_length == 0) {
1992  UPDATE_STATE(s_chunk_data_almost_done);
1993  }
1994 
1995  break;
1996  }
1997 
1998  case s_chunk_data_almost_done:
1999  assert(parser->flags & F_CHUNKED);
2000  assert(parser->content_length == 0);
2001  STRICT_CHECK(ch != CR);
2002  UPDATE_STATE(s_chunk_data_done);
2003  CALLBACK_DATA(body);
2004  break;
2005 
2006  case s_chunk_data_done:
2007  assert(parser->flags & F_CHUNKED);
2008  STRICT_CHECK(ch != LF);
2009  parser->nread = 0;
2010  UPDATE_STATE(s_chunk_size_start);
2011  CALLBACK_NOTIFY(chunk_complete);
2012  break;
2013 
2014  default:
2015  assert(0 && "unhandled state");
2016  SET_ERRNO(HPE_INVALID_INTERNAL_STATE);
2017  goto error;
2018  }
2019  }
2020 
2021  /* Run callbacks for any marks that we have leftover after we ran our of
2022  * bytes. There should be at most one of these set, so it's OK to invoke
2023  * them in series (unset marks will not result in callbacks).
2024  *
2025  * We use the NOADVANCE() variety of callbacks here because 'p' has already
2026  * overflowed 'data' and this allows us to correct for the off-by-one that
2027  * we'd otherwise have (since CALLBACK_DATA() is meant to be run with a 'p'
2028  * value that's in-bounds).
2029  */
2030 
2031  assert(((header_field_mark ? 1 : 0) +
2032  (header_value_mark ? 1 : 0) +
2033  (url_mark ? 1 : 0) +
2034  (body_mark ? 1 : 0) +
2035  (status_mark ? 1 : 0)) <= 1);
2036 
2037  CALLBACK_DATA_NOADVANCE(header_field);
2038  CALLBACK_DATA_NOADVANCE(header_value);
2039  CALLBACK_DATA_NOADVANCE(url);
2040  CALLBACK_DATA_NOADVANCE(body);
2041  CALLBACK_DATA_NOADVANCE(status);
2042 
2043  RETURN(len);
2044 
2045 error:
2046  if (HTTP_PARSER_ERRNO(parser) == HPE_OK) {
2047  SET_ERRNO(HPE_UNKNOWN);
2048  }
2049 
2050  RETURN(p - data);
2051 }
