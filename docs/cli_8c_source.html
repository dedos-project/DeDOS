509  cmd[ln] = '\0';

517  if ( strncasecmp(cmd, cmd_actions[i].cmd, strlen(cmd_actions[i].cmd)) == 0 ) {
518  if ( cmd_actions[i].action ) {
519  rtn = cmd_actions[i].action(cmd + strlen(cmd_actions[i].cmd) + 1 );

521  rtn = 1;
522  }
523  break;
524  }
525  }
526  if (rtn < 0){
